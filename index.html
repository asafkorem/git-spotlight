Git Spotlight

Discover insights in your git history with these carefully crafted command patterns. Each pattern helps you identify different aspects of your codebase.

Track frequently changing files that might need attention


# Find most frequently changed files
git log --since="6 months ago" --grep="fix|bug|refactor" -i --name-only --pretty=format: | \
  sort | \
  uniq -c | \
  sort -nr | \
  head -10

Dungeons

Find files with high activity but few contributors


# Get all unique files from git history
files=$(git log --since="1 year ago" --grep="fix|bug|refactor" -i --name-only --pretty=format: | sort -u)

# For each file, check changes and authors
echo "$files" | while read -r file; do
  if [ ! -z "$file" ]; then
    # Count changes
    changes=$(git log --since="1 year ago" --grep="fix|bug|refactor" -i --name-only --pretty=format: | grep -c "^$file" || echo 0)
    
    # If changes meet minimum threshold
    if [ "$changes" -ge 5 ]; then
      # Count unique authors
      author_count=$(git log --since="1 year ago" --pretty=format:"%an" -- "$file" | sort -u | wc -l)
      
      # If author count is within maximum
      if [ "$author_count" -le 2 ]; then
        echo "File: $file"
        echo "Changes: $changes"
        echo "Authors: "
        git log --since="1 year ago" --pretty=format:"%an" -- "$file" | sort | uniq -c | sort -nr | \
          awk '{
            count=$1
            author=$2
            for(i=3;i<=NF;i++) author=author " " $i
            printf "  %s: %d changes\n", author, count
          }'
        echo "---"
      fi
    fi
  fi
done
